建筑(buildings)在 /Hearts of Iron IV/common/buildings/*.txt 中定义。

buildings = {
    xxx = {
		base_cost = 4000 #基础建造花费
        per_level_extra_cost = 100 #每一级增加的建造花费（参考原版要塞）
		icon_frame = 3 #wiki：建筑物图标定义为 。整数显示 GFX_buildings_strip Sprite 中的哪个帧用于此建筑物。
        show_on_map = 模型
		infrastructure = yes #使建筑物被标记为基础设施。
		max_level = 5 #建造上限
		value = 1 #wiki：决定建筑物在空中轰炸行动中的基础生命值，该生命值会随着建筑物的每一层而成倍增加。此外，这也决定了国家在和平会议期间获得的价格或占据政治权力的成本
        base_cost_conversion = 3000 #转换花费
		show_modifier = yes #显示该建筑对地区的修正（应该）
		max_fuel_building = 1.5 #k （储存石油上限好像是）
		allied_build = yes #Wiki：如果指定，则使该建筑算作盟军建筑，使其修饰符不仅适用于建造它的国家，还适用于其盟友（即臣民、霸主和/或派系成员）。
        shares_slots = yes #是否和其他建筑共享该地区的格子（如军工民工共享一个地区的建筑槽）
        provincial = yes #是否只能在小格子上建造（如要塞），这些不能组合在一起（指shares_slots = yes）
		infrastructure_construction_effect = yes/no #是否受到基建等级的建造速度修正
        only_costal = yes/no #（原文如此），如果指定，则该建筑物只能在定义为沿海的省份或包含任何此类省份的州建造，具体取决于建筑物的类型。
        disabled_in_dmz = yes/no #如果添加，将使该建筑物无法在非军事区国家建造或使用。
        damage_factor = 0.3 #修改该建筑在空袭或陆战中受到的伤害。正值会增加它，而负值会减少它。
	}
    #原版模板（军工）
    arms_factory = {
		show_on_map = 6
		base_cost = 6000
		base_cost_conversion = 3000
		military_production = 1
		icon_frame = 2

	}
    
}

#每个建筑物都可以接受其自身内部的任何状态范围的修饰符，并在构建时将其应用于状态。基础游戏中使用的一些包括 、（用于防空）和（用于核反应堆）。对于省级建筑物，则接受省级修饰符。这也可以与修饰符标记一起使用，以创建作为变量系统的自定义效果。添加将使修饰符显示在建筑物的工具提示中。
#local_resources_<resource> air_defencenuclear_production_factorshow_modifier = yes
#除了状态范围的修饰符之外，还有其他几个参数可以使建筑物具有与基础游戏建筑物相同的效果，或者被视为 AI 的效果：

### 人话：ai会以为这个建筑是什么 ###

military_production = 将指定数量的军工厂添加到控制器中（wiki笔记：大于1的值将与1工厂相同，但可能介于0和1之间）
general_production = 将指定数量的民用工厂添加到控制器中（wiki笔记：大于1的值将与1工厂相同，但可能介于0和1之间）
naval_production = 将指定数量的船坞添加到控制器（wiki笔记：大于1的值将与1工厂相同，但可能介于0和1之间）
infrastructure =  yes/no使建筑物被标记为基础设施（包括施工速度加成和额外资源）
air_base = yes/no使建筑物被标记为空军基地
supply_node = yes/no使建筑物被标记为供应节点
is_port = yes/no使建筑物被标记为海军港口
land_fort = 数字，为该省增加了许多陆地堡垒
naval_fort = 数字，为该省增加了许多沿海堡垒
refinery = yes/no使建筑物被标记为合成精炼厂
fuel_silo = yes/no使建筑物被标记为燃料筒仓
radar = yes/no使建筑物被标记为雷达站
rocket_production = 数字，定义每天在火箭生产方面取得了多少进展
rocket_launch_capacity = 数字，定义构建每天可以发射多少枚火箭
nuclear_reactor =  yes/no使建筑物被标记为核反应堆
